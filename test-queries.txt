***** View the list of children *****
SELECT * FROM Child;

***** View the list of parents *****
SELECT * FROM Parent;

***** Select all children of parents *****
SELECT p.first_name, p.last_name, c.first_name, c.last_name 
FROM Parent p, Child c, ParentHasChild phs,
WHERE p.parent_id = phs.parent_id;
AND c.child_id = phs.child_id;
